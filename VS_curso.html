Gráfico de puntos y de líneas

alt.Chart(movies).mark_point().encode(
    alt.X('runtime')
)

alt.Chart(movies).mark_tick().encode(
    alt.X('runtime')
)

Histograma

alt.Chart(movies).mark_bar().encode(
    alt.X('runtime', bin = alt.Bin(maxbins = 30)),
    alt.Y('count()')
)

Gráfico de densidad

alt.Chart(movies).transform_density(
    'runtime',
    as_ = ['runtime', 'density']
).mark_area().encode(
    alt.X('runtime:Q'),
    alt.Y('density:Q')
)
alt.Chart(movies).transform_density(
    'runtime',
    groupby = ['genre'],
    as_ = ['runtime', 'density']
).mark_area().encode(
    alt.X('runtime:Q'),
    alt.Y('density:Q', stack = None),
    alt.Color('genre')
)
